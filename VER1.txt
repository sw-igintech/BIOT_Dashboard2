    function isoDateOnly(isoString)
    {
      /* Convert ISO timestamp -> YYYY-MM-DD (in local time display) */
      const d = new Date(isoString);
      const yyyy = d.getFullYear();
      const mm = String(d.getMonth() + 1).padStart(2, "0");
      const dd = String(d.getDate()).padStart(2, "0");
      return `${yyyy}-${mm}-${dd}`;
    }

    function formatDisplayDateTime(isoString)
    {
      /* Format for table display: DD/MM/YYYY HH:mm:ss (like the screenshot) */
      const d = new Date(isoString);
      const dd = String(d.getDate()).padStart(2, "0");
      const mm = String(d.getMonth() + 1).padStart(2, "0");
      const yyyy = d.getFullYear();
      const hh = String(d.getHours()).padStart(2, "0");
      const mi = String(d.getMinutes()).padStart(2, "0");
      const ss = String(d.getSeconds()).padStart(2, "0");
      return `${dd}/${mm}/${yyyy} ${hh}:${mi}:${ss}`;
    }

    function getDefaultRangeFromData()
    {
      /* Pick default date filter bounds based on existing events */
      const dates = gloveEvents.map(e => new Date(e.ts).getTime()).sort((a,b) => a-b);
      const min = new Date(dates[0]);
      const max = new Date(dates[dates.length - 1]);

      /* Use YYYY-MM-DD for input[type=date] */
      const from = isoDateOnly(min.toISOString());
      const to = isoDateOnly(max.toISOString());

      return { from, to };
    }

    function filterGloveEvents(fromDateStr, toDateStr)
    {
      /* Inclusive filtering on date (not time) */
      const from = new Date(`${fromDateStr}T00:00:00`).getTime();
      const to = new Date(`${toDateStr}T23:59:59`).getTime();

      return gloveEvents.filter(e =>
      {
        const t = new Date(e.ts).getTime();
        return t >= from && t <= to;
      });
    }

    function sumGlovesByType(events)
    {
      /* Aggregate quantities by glove type */
      const totals = { "Small": 0, "Medium": 0, "Large": 0, "Extra Large": 0 };

      for (const e of events)
      {
        if (totals[e.gloveType] === undefined)
        {
          totals[e.gloveType] = 0;
        }
        totals[e.gloveType] += e.qty;
      }
      return totals;
    }

    function renderLegend(containerEl, labels, values, colors, total)
    {
      /* Manual legend like the screenshot: label on left, percentage on right */
      containerEl.innerHTML = "";

      for (let i = 0; i < labels.length; i++)
      {
        const label = labels[i];
        const value = values[i];
        const pct = total > 0 ? Math.round((value / total) * 100) : 0;

        const row = document.createElement("div");
        row.className = "legend-item";

        const left = document.createElement("div");
        left.className = "legend-left";

        const sw = document.createElement("span");
        sw.className = "swatch";
        sw.style.background = colors[label] || "#999";

        const txt = document.createElement("span");
        txt.textContent = label;

        left.appendChild(sw);
        left.appendChild(txt);

        const right = document.createElement("div");
        right.textContent = `${pct}%`;

        row.appendChild(left);
        row.appendChild(right);

        containerEl.appendChild(row);
      }
    }

    function initGloveChart()
    {
      const ctx = document.getElementById("gloveChart");
      const legendEl = document.getElementById("gloveLegend");

      const range = getDefaultRangeFromData();
      document.getElementById("fromDate").value = range.from;
      document.getElementById("toDate").value = range.to;

      const filtered = filterGloveEvents(range.from, range.to);
      const totals = sumGlovesByType(filtered);

      const labels = Object.keys(totals);
      const values = labels.map(k => totals[k]);
      const totalValue = values.reduce((a,b) => a + b, 0);

      document.getElementById("gloveTitle").textContent = `Glove Consumption (Total: ${totalValue})`;
      document.getElementById("gloveSubtitle").textContent =
        `Filter: ${range.from} 00:00:00 to ${range.to} 23:59:59`;

      renderLegend(legendEl, labels, values, gloveColors, totalValue);

      gloveChart = new Chart(ctx,
      {
        type: "pie",
        data:
        {
          labels,
          datasets:
          [{
            data: values,
            backgroundColor: labels.map(l => gloveColors[l] || "#999"),
            borderColor: "#ffffff",
            borderWidth: 2
          }]
        },
        options:
        {
          responsive: true,
          maintainAspectRatio: false,
          plugins:
          {
            legend: { display: false },
            tooltip:
            {
              callbacks:
              {
                label: (ctx) =>
                {
                  const v = ctx.raw;
                  const pct = totalValue > 0 ? ((v / totalValue) * 100).toFixed(1) : "0.0";
                  return ` ${ctx.label}: ${v} (${pct}%)`;
                }
              }
            }
          }
        }
      });

      /* Give the canvas a stable height so the card looks like the screenshot */
      ctx.parentElement.style.height = "300px";
    }

    function applyGloveFilter()
    {
      const from = document.getElementById("fromDate").value;
      const to = document.getElementById("toDate").value;

      const filtered = filterGloveEvents(from, to);
      const totals = sumGlovesByType(filtered);

      const labels = Object.keys(totals);
      const values = labels.map(k => totals[k]);
      const totalValue = values.reduce((a,b) => a + b, 0);

      document.getElementById("gloveTitle").textContent = `Glove Consumption (Total: ${totalValue})`;
      document.getElementById("gloveSubtitle").textContent =
        `Filter: ${from} 00:00:00 to ${to} 23:59:59`;

      renderLegend(document.getElementById("gloveLegend"), labels, values, gloveColors, totalValue);

      /* Update chart in-place */

      gloveChart.data.datasets[0].data = values;
      gloveChart.data.datasets[0].backgroundColor = labels.map(l => gloveColors[l] || "#999");
      gloveChart.update();
    }

    {
      const range = getDefaultRangeFromData();
      document.getElementById("fromDate").value = range.from;
      document.getElementById("toDate").value = range.to;
      applyGloveFilter();
